const ds = document.getElementById("display")

for (i = 0; i < localStorage.length; i++) {
    const kei = localStorage.key(i)
    const valu = localStorage.getItem(`${kei}`)
    const newText = document.createElement("h1")
    newText.innerHTML = kei + ": " + valu
    console.log(valu)
    ds.appendChild(newText)
}
